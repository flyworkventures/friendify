
server {
    listen 80;
    listen [::]:80;
    server_name friendfy.fly-work.com;

    # Certbot için challenge izinleri (Let’s Encrypt yenileme için gerekli)
    location /.well-known/acme-challenge/ {
        root /var/www/html;
    }

    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name friendfy.fly-work.com;

    # SSL Sertifikaları
    ssl_certificate /etc/letsencrypt/live/friendfy.fly-work.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/friendfy.fly-work.com/privkey.pem;

    # Güvenlik ayarları
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers on;

    # Proxy ayarları
    location / {
        proxy_pass http://127.0.0.1:3001;   # Node.js API bunlara gidiyor

        # Websocket desteği (chat için çok önemli)
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        # Normal header'lar
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
